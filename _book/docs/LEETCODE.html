  1. 数组去重
var removeDuplicates = function(nums) {
    var a = {}, b = [];
    for(var i = 0; i < nums.length; i ++){
        var key = nums[i].toString();
        if(!a[key]){
          a[key] = true;
          b.push(nums[i])
         }
    }
    return b
};
//removeDuplicates([1,2,1,2,4]) => [1,2,4]
