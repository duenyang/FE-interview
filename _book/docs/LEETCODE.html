function Fibonacci(index){
    if (index < 0) {
        return 0;
    }
    if (index <= 2) {
        return 1;
    }
    return Fibonacci(index - 1) + Fibonacci(index - 2);
}

var removeDuplicates = function(nums) {
    var a = {}, b = [];
    for(var i = 0; i < nums.length; i ++){
        var key = nums[i].toString();
        if(!a[key]){
          a[key] = true;
          b.push(nums[i])
        }
    }
    return b
};
